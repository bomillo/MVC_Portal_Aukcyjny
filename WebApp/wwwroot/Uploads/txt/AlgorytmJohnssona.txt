Johnson(G) 
{
	Oblicz G’, gdzie E[G’]==E[G]+{ (s, v): vcV[G] }
	if(!Bellman-Ford(G’, w, s))
	wypisz “graf wejściowy zawiera cykl o ujemnej wadze”;
	else 
	{
		for(każdego vcV[G’])
		h(v) = delta(s, v), gdzie wartość delata(s, v) obliczono w algorytmie Bellmana-Forda;
		for(każdej krawędzi (u, v)cE[G’]) wˆ(u, v) = w(u, v)+h(u)-h(v);
		for(każdego wierzchołka ucV[G] 
		{
			wywołaj Dijkstry(G, wˆ , u), aby obliczyć deltaˆ(u, v) dla każdego vcV[G];
			for(każdego wierzchołka vcV[G])
			D[u][v]=deltaˆ(u, v)+h(v)-h(u);
			Zapisz macierz D; 
		}
	}
}
